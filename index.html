“Define your scaffolding”

Part 1

Templates are parsed but unrendered DOM trees which can be reused. 
            <template id="welcome">
                <p>Welcome!</p>
                <script>
                    alert("I've been activated");
                </script>
            </template> 
            var tpl = document.querySelector("#welcome");
            var clone = tpl.content.cloneNode(true);
            // Still no alert…
            document.querySelector('something').appendChild(clone);
            // Alert!

“Controllably apply content & styles”

Styling from the outside

            <template>
                <h1 part="title">Some Title…</h1>
                <h2 part="subtitle">… with a subtitle</h2>
            </template>
            var tmpl = document.querySelector("template");
            var host = document.querySelector(".widget");
            var root = host.createShadowRoot();
            root.appendChild(tmpl.content.cloneNode(true)); 
          .widget::part(title) { font-family: Courier; }
          .widget::part(subtitle) { font-style: italic; }
  • Only descendants of Shadow Root will be rendered

“Define an API”

            var tpl = document.querySelector("#some-template");
            var SpoilerPrototype = Object.create(HTMLElement.prototype);

            SpoilerPrototype.createdCallback = function() {
                var root = this.createShadowRoot();
                root.appendChild(tpl.content.cloneNode(true));
            };

            var Spoiler = document.register("spoiler-tag", {
                prototype: SpoilerPrototype
            });
  • Tag names must contain a dash

Polymer

All in one step

            <polymer-element name="my-element">
                <template>
                    <h2>
                        Hello, I'm a Polymer element
                    </h2>
                </template>
                <script>
                    Polymer("my-element", {});
                </script>
            </polymer-element> 
            <my-element></my-element>

Polymer

Templating

            <polymer-element name="fav-color" attributes="color">
                <template>
                    My favorite color is
                    <span style="background: {{ color }}">{{ color }}</span>
                    <input type="text" value="{{ color }}">
                </template>
                <script>
                    Polymer("fav-color", {
                        color: "Orange"
                    });
                </script>
            </polymer-element>
